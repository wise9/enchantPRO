 55     enchant.pro.Sprite3D = enchant.Class.create(enchant.EventTarget, {
 56 	    /**
 57 	     * 3Dポリゴン表示機能を持ったクラス.
 58 	     *
 59 	     * @example
 60 	     *   var bear = new Sprite();
 61 	     *   bear.image = game.assets['chara1.gif'];
 62 	     *
 63 	     * @constructs
 64 	     * @extends enchant.EventTarget
 65 	     */
 66         initialize: function() {
 67             enchant.EventTarget.call(this);
 68 
 69             this._mesh = EP_GL.getFactory().createMesh();
 70             this._meshId = this._mesh.getMeshId();
 71             enchant.pro.Sprite3DTable[this._meshId] = this;
 72             this._x = 0;
 73             this._y = 0;
 74             this._z = 0;
 75             this.childNodes = [];
 76             this.scene = null;
 77 
 78             var listener = function(e) {
 79                 for (var i = 0, len = this.childNodes.length; i < len; i++) {
 80                     var sprite = this.childNodes[i];
 81                     sprite.dispatchEvent(e);
 82                 }
 83             };
 84             this.addEventListener('added', listener);
 85             this.addEventListener('removed', listener);
 86             this.addEventListener('addedtoscene', listener);
 87             this.addEventListener('removedfromscene', listener);
 88 
 89 			var that = this;
 90             var render = function() {
 91                 that.dispatchEvent(new enchant.Event('render'));
 92             };
 93             this.addEventListener('addedtoscene', function() {
 94                 render();
 95                 game.addEventListener('exitframe', render);
 96             });
 97             this.addEventListener('removedfromscene', function() {
 98                 game.removeEventListener('exitframe', render);
 99             });
100 
101             this.addEventListener('render', function() {
102                 if (this._changedTranslation) {
103                     this._mesh.setTranslationJSON(JSON.stringify([this._x, this._y, this._z]));
104                     this._changedTranslation = false;
105                 }
106                 if (this._changedScale) {
107                     this._mesh.setScaleJSON(JSON.stringify([this._scaleX, this._scaleY, this._scaleZ]));
108                     this._changedScale = false;
109                 }
110             });
111         },
112 	    /**
113 	     * 子ポリゴンを追加する.
114 	     * @param {Sprite3D} sprite 追加する子ポリゴン.
115 	     */
116         addChild: function(sprite) {
117             this.childNodes.push(sprite);
118             this._mesh.addChild(sprite._mesh);
119             sprite.parentNode = this;
120             sprite.dispatchEvent(new enchant.Event('added'));
121             if (this.scene) {
122                 sprite.scene = this.scene;
123                 sprite.dispatchEvent(new enchant.Event('addedtoscene'));
124             }
125         },
126 	    /**
127 	     * 指定された子ポリゴンを削除する.
128 	     * @param {Sprite3D} sprite 削除する子ポリゴン.
129 	     */
130         removeChild: function(sprite) {
131             var i;
132             if ((i = this.childNodes.indexOf(sprite)) != -1) {
133                 this.childNodes.splice(i, 1);
134             }
135             this._mesh.removeChild(sprite._mesh);
136             sprite.parentNode = null;
137             sprite.dispatchEvent(new enchant.Event('removed'));
138             if (this.scene) {
139                 sprite.scene = null;
140                 sprite.dispatchEvent(new enchant.Event('removedfromscene'));
141             }
142         },
143         /**
144          * Sprite3Dを平行移動する.
145          * @param {Number} x
146          * @param {Number} y
147          * @param {Number} z
148          */
149         translate: function(x, y, z) {
150             this._x += x;
151             this._y += y;
152             this._z += z;
153             this._changedTranslation = true;
154         },
155         /**
156          * Sprite3Dを拡大縮小する.
157          * @param {Number} x
158          * @param {Number} y
159          * @param {Number} z
160          */
161         scale: function(x, y, z) {
162             this._scaleX *= x;
163             this._scaleY *= y;
164             this._scaleZ *= z;
165             this._changedScale = true;
166         },
167         /**
168          * Sprite3Dの回転行列
169          * @type {Array.<Number>}
170          */
171         rotation: {
172             get: function() {
173                 return this._rotation;
174             },
175             set: function(rotation) {
176                 this._rotation = rotation;
177                 this._mesh.setRotationJSON(JSON.stringify(rotation));
178             }
179         },
180         /**
181          * Sprite3Dの変換行列
182          * @type {Array.<Number>}
183          */
184         matrix: {
185             get: function() {
186                 return this._matrix;
187             },
188             set: function(matrix) {
189                 this._matrix = matrix;
190                 this._mesh.setMatrixJSON(JSON.stringify(matrix));
191             }
192         },
193         /**
194          * Sprite3Dのテクスチャ
195          * @type {Surface}
196          */
197         surface: {
198             get: function() {
199                 return this._surface;
200             },
201             set: function(surface) {
202                 this._surface = surface;
203                 this._mesh.setMaterial(this._surface._material);
204             }
205         },
206         /**
207          * Sprite3Dの名前
208          * @type {String}
209          */
210         name: {
211             get: function() {
212                 return this._name;
213             },
214             set: function(surface) {
215                 this._mesh.setName(this._name = name);
216             }
217         }

271 	    /**
272 	     * 表示ポリゴンツリーのルートになるクラス.
273 	     *
274 	     * @example
275 	     *   var scene = new Scene3D();
276 	     *   var sprite = new Sprite3D();
277 	     *   scene.addChild(sprite);
278 	     *
279 	     * @constructs
280 	     * @extends enchant.EventTarget
281 	     */
282 	     initialize: function() {
283             enchant.EventTarget.call(this);
284 
285             this._manager = EP_GL.getMeshManager();
286             this.childNodes = [];
287             this.lights = [];
288             var that = this;
289             enchant.Game.instance.addEventListener('enterframe', function(e) {
290                 var nodes = that.childNodes.slice();
291                 var push = Array.prototype.push;
292                 while (nodes.length) {
293                     var node = nodes.pop();
294                     node.dispatchEvent(e);
295                     if (node.childNodes) {
296                         push.apply(nodes, node.childNodes);
297                     }
298                 }
299             	if(that._camera){
300             		if(that._camera._changedPosition){
301             			EP_GL.getCamera().setPositionJSON(
302             				"[" + that._camera.x + "," + that._camera.y + "," + that._camera.z + "]");
303             			that._camera._changedPosition = false;
304             		}
305             		if(that._camera._changedCenter){
306 	            		EP_GL.getCamera().setCenterJSON(
307 	            			"[" + that._camera.centerX + "," + that._camera.centerY + "," + that._camera.centerZ + "]");
308    						that._camera._changedCenter = false;
309             		}
310             		if(that._camera._changedUpVector){
311 	            		EP_GL.getCamera().setUpVectorJSON(
312 	            			"[" + that._camera.upVectorX + "," + that._camera.upVectorY + "," + that._camera.upVectorZ + "]");
313 	            		that._camera._changedUpVector = false;
314             		}
315             	}
316             	nodes = that.lights.slice();
317             	while(nodes.length) {
318             		var node = nodes.pop();
319             		if(node._changedPosition){
320             			EP_GL.getLight(node._id).setPositionJSON(
321             				"[" + node.x + "," + node.y + "," + node.z + "]");
322             			node._changedPosition = false;
323             		}
324             		if(node._changedDirection){
325             			EP_GL.getLight(node._id).setSpotDirectionJSON(
326             				"[" + node.directionX + "," + node.directionY + "," + node.directionZ + "]");
327             			node._changedDirection = false;
328             		}
329             	}
330             });
331         },
332         /**
333          * Scene3Dにポリゴンを追加する.
334          * @param {enchant.Sprite3D}
335          */
336         addChild: function(sprite) {
337         	this.childNodes.push(sprite);
338             this._manager.add(sprite._mesh);
339             sprite.parentNode = sprite.scene = this;
340             sprite.dispatchEvent(new enchant.Event('added'));

356         },

544     enchant.pro.Sensor = enchant.Class.create(enchant.EventTarget, {
545     	/**
546     	 * センサークラス
547     	 * @constructs
548     	 */
549     	initialize: function(className){
550     		if (enchant.pro[className].instance) {
551     			return enchant.pro[className].instance;
552     		}
553     		if(this instanceof enchant.pro[className]) {
554     			enchant.pro[className].instance = this;
555     		} else {
556     			return new enchant.pro[className];
557     		}
558     		enchant.EventTarget.call(this);
559 	    	var sensorList = EP_SENSOR.getSensor(className).getSensorsList();
560 	    	if(sensorList.size() > 0){
561 	    		this._sensor = sensorList.get(0);
562 	    	}
563 	    	if(this._sensor){
564 	    		this._sensor.setCallback("enchant.pro." + className + ".onUpdate");
565 	    	}
566 	    	
567 	    	enchant.pro[className].onUpdate = function(data){
568 		        if (enchant.pro[className].instance) {
569 		            var e = new Event('update');
570 		            e.data = data;
571 		            enchant.pro[className].instance.dispatchEvent(e);
572 		        }
573 	    	};
574     	},
575     	/**
576    		 * センサーが利用可能かを返す
577    		 * @return {Boolean} 
578    		 */
579    		isAvailable: function(){
580    			return this._sensor;
581    		},
582    		/**
583    		 * センサーからの読み取りを開始
584    		 */
585    		start: function(){
586    			if(this._sensor){
587     			this._sensor.enable();
588    			}
589    		},
590    		/**
591    		 * センサーからの読み取りを終了
592    		 */
593    		stop: function(){
594    			if(this._sensor){
595    				this._sensor.disable();
596    			}
597    		}
598    	});

864     var touches = [null, null, null, null, null];
865     ['start', 'move', 'end'].forEach(function(type) {
866         window['androidTouch' + capitalize(type)] = function(touch) {
867         	if (touches[touch.identifier]) {
868                 touch.target = touches[touch.identifier].target;    
869             } else {
870                 touch.target = document.elementFromPoint(touch.pageX, touch.pageY);
871             }
872 
873             if (type == 'end') {
874                 touches[touch.identifier] = null;
875             } else {
876                 touches[touch.identifier] = touch;
877             }
878 
879             var e = document.createEvent('MouseEvent');
880             e.emulated = true;
881             
882             e.initMouseEvent('touch' + type, true, false, window, 0,
883                 touch.pageX, touch.pageY, touch.pageX, touch.pageY, false, false, true, false, 0, null);
884             e.touches=touches.filter(function(i) { return i && i.target == this; }, touch.target);
885             e.changedTouches = [touch];
886             touch.target.dispatchEvent(e);
887         };
